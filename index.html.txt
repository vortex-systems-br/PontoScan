// ── STORAGE ──
const ls = {
  get: (k, fb = null) => { try { const v = localStorage.getItem(k); return v ? JSON.parse(v) : fb; } catch { return fb; } },
  set: (k, v) => { try { localStorage.setItem(k, JSON.stringify(v)); } catch {} },
  del: (k) => { try { localStorage.removeItem(k); } catch {} },
};
const SK = { USERS:"ps_users", SESSION:"ps_session", HISTORY:"ps_history", API_KEY:"ps_gemini_key" };

// ── AUTH ──
const encode = s => btoa(unescape(encodeURIComponent(s)));
function register({ name, email, password }) {
  const users = ls.get(SK.USERS, []);
  if (users.find(u => u.email.toLowerCase() === email.trim().toLowerCase()))
    return { error: "E-mail já cadastrado. Faça login." };
  const user = { id: Date.now().toString(), name: name.trim(), email: email.trim().toLowerCase(), hash: encode(password) };
  ls.set(SK.USERS, [...users, user]);
  return { user };
}
function login({ email, password }) {
  const users = ls.get(SK.USERS, []);
  const user = users.find(u => u.email.toLowerCase() === email.trim().toLowerCase());
  if (!user) return { error: "E-mail não encontrado." };
  if (user.hash !== encode(password)) return { error: "Senha incorreta." };
  return { user };
}

}
function useClock() {
  const fmt = () => { const d = new Date(); return `${String(d.getHours()).padStart(2,"0")}:${String(d.getMinutes()).padStart(2,"0")}`; };
  const [t, setT] = useState(fmt);
  useEffect(() => { const id = setInterval(() => setT(fmt()), 15000); return () => clearInterval(id); }, []);
  return t;
}

// ── IMAGE ──
function compressImage(file, maxW = 1400, q = 0.82) {
  return new Promise((res, rej) => {
    const img = new Image();
    const url = URL.createObjectURL(file);
    img.onload = () => {
      URL.revokeObjectURL(url);
      let { width: w, height: h } = img;
      if (w > maxW) { h = Math.round(h * maxW / w); w = maxW; }
      const c = document.createElement("canvas");
      c.width = w; c.height = h;
      c.getContext("2d").drawImage(img, 0, 0, w, h);
      c.toBlob(b => b ? res(b) : rej(new Error("Compressão falhou")), "image/jpeg", q);
    };
    img.onerror = () => { URL.revokeObjectURL(url); rej(new Error("Imagem inválida")); };
    img.src = url;
  });
}
function blobToBase64(blob) {
  return new Promise((res, rej) => {
    const r = new FileReader();
    r.onload = () => res(r.result.split(",")[1]);
    r.onerror = () => rej(new Error("Erro de leitura"));
    r.readAsDataURL(blob);
  });
}

Retorne SOMENTE JSON válido sem markdown nem texto extra:

async function analyzeWithGemini(base64, apiKey, onStep) {
  if (!apiKey?.trim()) throw new Error("Chave API não configurada. Vá em Configurações → Chave Gemini API.");
  onStep(1);
  let res;
  try {
    res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey.trim()}`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ contents: [{ parts: [{ inline_data: { mime_type: "image/jpeg", data: base64 } }, { text: PROMPT }] }], generationConfig: { temperature: 0.1, maxOutputTokens: 2048 } })
    });
  } catch { throw new Error("Sem conexão. Verifique sua internet."); }
  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    const msg = err?.error?.message || "";
    if (res.status === 400) throw new Error("Chave API inválida. Verifique em Configurações.");
    if (res.status === 403) throw new Error("Chave sem permissão. Ative a Gemini API no Google AI Studio.");
    if (res.status === 429) throw new Error("Limite atingido. Aguarde 1 minuto.");
    throw new Error(msg || `Erro ${res.status}.`);
  }
  onStep(2);
  const data = await res.json();
  if (data.candidates?.[0]?.finishReason === "SAFETY") throw new Error("Imagem bloqueada por filtros. Tente outra.");
  const text = data.candidates?.[0]?.content?.parts?.[0]?.text || "";
  if (!text) throw new Error("Resposta vazia. Tente novamente.");
  onStep(3);
  const match = text.match(/\{[\s\S]*\}/);
  if (!match) throw new Error("Formato inesperado. Tente novamente.");
  let parsed;
  try { parsed = JSON.parse(match[0]); } catch { throw new Error("Erro ao interpretar. Tente novamente."); }
  if (parsed.error) throw new Error(parsed.error);
  return parsed;
}

// ── PROCESSING ──
const STEPS = [
  { l:"Comprimindo imagem", s:"Otimizando para análise..." },
  { l:"OCR com Gemini Vision", s:"Lendo texto e tabelas..." },
  { l:"Organizando registros", s:"Estruturando horários..." },
  { l:"Calculando totais", s:"Verificando inconsistências..." },
];
function ProcessingScreen({ file, apiKey, onDone, onError }) {
  const [step, setStep] = useState(-1);
  const [pct, setPct] = useState(0);
  const abort = useRef(false);
  const ran = useRef(false);

  const run = useCallback(async () => {
    if (!file || ran.current) return;
    ran.current = true; abort.current = false;
    try {
      setStep(0); setPct(10);
      const blob = await compressImage(file);
      if (abort.current) return;
      const b64 = await blobToBase64(blob);
      if (abort.current) return;
      const result = await analyzeWithGemini(b64, apiKey, s => {
        if (abort.current) return;
        setStep(s);
        setPct(Math.round((s + 1) / STEPS.length * 85) + 10);
      });
      if (abort.current) return;
      setStep(STEPS.length); setPct(100);
      setTimeout(() => { if (!abort.current) onDone(result, file); }, 500);
    } catch(e) {
      if (!abort.current) onError(e.message || "Erro inesperado.");
    }
  }, [file, apiKey, onDone, onError]);

  useEffect(() => { run(); return () => { abort.current = true; }; }, [run]);

  return (
    <div className="scroll">
      <SBar />
      <div style={{ padding:"8px 20px 100px", display:"flex", flexDirection:"column", alignItems:"center" }}>
        <div style={{ width:90, height:90, position:"relative", margin:"24px auto 24px", display:"flex", alignItems:"center", justifyContent:"center" }}>

            <I.Brain />
          </div>
        </div>
        <h1 className="f1" style={{ fontFamily:"'Syne',sans-serif", fontSize:18, fontWeight:800, color:"#E2E8F7", letterSpacing:"-0.03em", textAlign:"center", marginBottom:5 }}>

// ── TOAST ──
function Toast({ msg, onClose, onRetry }) {
  useEffect(() => { const t = setTimeout(onClose, 9000); return () => clearTimeout(t); }, [onClose]);
  return (
    <div className="toast">
      <div style={{ display:"flex", justifyContent:"space-between", gap:10, marginBottom:onRetry?9:0 }}>
        <p style={{ flex:1, lineHeight:1.55 }}>{msg}</p>
        <button onClick={onClose} style={{ background:"none", border:"none", color:"#FCA5A5", cursor:"pointer", fontSize:18, lineHeight:1 }}>✕</button>
      </div>
      {onRetry && (
        <button className="btn btn-danger" style={{ width:"100%", height:34, fontSize:13 }} onClick={onRetry}>
          Tentar novamente
        </button>
      )}
    </div>
  );
}

// ── APP ──
export default function App() {
  const [session, setSession] = useState(() => ls.get(SK.SESSION));
  const [auth, setAuth] = useState("login");
  const [apiKey, setApiKey] = useState(() => ls.get(SK.API_KEY, ""));
  const [nav, dispatch] = useReducer(navReducer, { prev:null, current:"home", dir:"r" });
  const [pFile, setPFile] = useState(null);
  const [result, setResult] = useState(null);
  const [rFile, setRFile] = useState(null);
  const [errMsg, setErrMsg] = useState(null);
  const [history, setHistory] = useState(() => ls.get(SK.HISTORY, []));

  const goTo = useCallback(t => dispatch({ type:"GO", to:t }), []);

  const handleLogin = useCallback(sess => {
    setSession(sess);
    setHistory(ls.get(SK.HISTORY, []));
    goTo("home");
  }, [goTo]);

  const handleLogout = useCallback(() => {
    setSession(null); setPFile(null); setResult(null); setErrMsg(null);
    setAuth("login");
  }, []);

  const handleApiKey = useCallback(k => {
    setApiKey(k);
    ls.set(SK.API_KEY, k);
  }, []);

  const handleFile = useCallback(file => {
    setPFile(file); setErrMsg(null); goTo("processing");
  }, [goTo]);

  const handleDone = useCallback((data, file) => {
    setResult(data); setRFile(file);
    const name = data?.employee?.name || "Não identificado";
    const now = new Date();
    const ms = ["Jan","Fev","Mar","Abr","Mai","Jun","Jul","Ago","Set","Out","Nov","Dez"];
    const entry = { name, date:`${now.getDate()} ${ms[now.getMonth()]}`, hours:data?.summary?.total_hours||"—", ts:now.getTime() };
    setHistory(h => { const n = [entry, ...h].slice(0,50); ls.set(SK.HISTORY, n); return n; });
    goTo("result");
  }, [goTo]);

  const handleErr = useCallback(msg => { setErrMsg(msg); goTo("capture"); }, [goTo]);
  const handleRetry = useCallback(() => { setErrMsg(null); if (pFile) goTo("processing"); }, [pFile, goTo]);

  const ss = useCallback(id => {
    if (id === nav.current) return "on";
    return SCREEN_ORDER.indexOf(id) < SCREEN_ORDER.indexOf(nav.current) ? "l" : "r";
  }, [nav.current]);

  return (
    <>
      <style>{css}</style>
      <div className="aw">
        <div className="shell">
          <div className="notch" />
          {!session ? (
            <>
              {auth === "login" && <div className="scr" data-s="on"><LoginScreen onLogin={handleLogin} onReg={() => setAuth("register")} /></div>}
              {auth === "register" && <div className="scr" data-s="on"><RegScreen onLogin={handleLogin} onBack={() => setAuth("login")} /></div>}
            </>
          ) : (
            <>
              <div className="scr" data-s={ss("home")}>
                <HomeScreen onCapture={() => goTo("capture")} history={history} session={session} />
                <BNav cur={nav.current} onNav={goTo} />
              </div>
              <div className="scr" data-s={ss("capture")}>
                <CaptureScreen onBack={() => goTo("home")} onProcess={handleFile} apiKey={apiKey} />
              </div>
              <div className="scr" data-s={ss("processing")}>
                {nav.current === "processing" && pFile && (
                  <ProcessingScreen file={pFile} apiKey={apiKey} onDone={handleDone} onError={handleErr} />
                )}
              </div>
              <div className="scr" data-s={ss("result")}>
                {result && <ResultScreen data={result} imageFile={rFile} onBack={() => goTo("capture")} onHome={() => goTo("home")} />}
                <BNav cur={nav.current} onNav={goTo} />
              </div>
              <div className="scr" data-s={ss("company")}>
                <CompanyScreen />
                <BNav cur={nav.current} onNav={goTo} />
              </div>
              <div className="scr" data-s={ss("settings")}>
                <SettingsScreen session={session} apiKey={apiKey} onApiKey={handleApiKey} onLogout={handleLogout} />
                <BNav cur={nav.current} onNav={goTo} />
              </div>
              {errMsg && <Toast msg={errMsg} onClose={() => setErrMsg(null)} onRetry={pFile ? handleRetry : null} />}
            </>
          )}
        </div>
      </div>
    </>
  );
}
